javascript

$(function(){
    var Y,M,D,H;
    function show(){
        var mydate = new Date();
        Y = mydate.getFullYear()+"-";
        M = (mydate.getMonth()+1)+"-";
        D = mydate.getDate();
        H=mydate.toLocaleTimeString();

    }
    show();
    //prepend()---在文档的开头插入内容
    $("#btn").click(function(){
          if($('textarea').val()!=""){
              $(".timeline").prepend("<li><span></span><div><p>"+Y+M+D+"</p><a href=''>"+$('textarea').val()+"</a></div></li>")
          }else{
              alert("请输入文字")
          }

    });
})
